 11def powerset(iterable: Iterable, max_size: int = None) -> Iterable:
 12    """Enumerate all _non-empty_ subsets of an iterable up to a given maximum size, e.g.:
 13    powerset([1,2,3]) --> (1,) (2,) (3,) (1,2) (1,3) (2,3) (1,2,3)
 14
 15    lightly adapted from itertools Recipes at
 16    https://docs.python.org/3/library/itertools.html#itertools-recipes
 17
 18    Args:
 19        iterable: elements from which to form subsets
 20        max_size: largest subsets (inclusive) to return
 21
 22    Returns:
 23        iterator over all subsets from size 1 to `max_size` of elements from `iterable`
 24    """
 25    s = list(iterable)
 26    if max_size is None:
 27        max_size = len(s)
 28    return chain.from_iterable(combinations(s, r) for r in range(1, max_size + 1))

 31def all_meanings(universe: Universe) -> Generator[Meaning, None, None]:
 32    """Generate all Meanings (sets of Referents) from a given Universe."""
 33    referents = universe.referents
 34    for refset in powerset(referents):
 35        yield Meaning(refset, universe)

 64def upto_comb(num: int, max_k: int) -> int:
 65    """Return the number of ways of choosing _up to max_k_ items from
 66    n items without repetition.  Just an iterator of math.comb for n from
 67    1 to max_k."""
 68    return sum(comb(num, k) for k in range(1, max_k + 1))

 71def random_languages(
 72    expressions: Iterable[Expression],
 73    sampling_strategy: str = "uniform",
 74    sample_size: int = None,
 75    language_class: Type[Language] = Language,
 76    max_size: int = None,
 77) -> list[Language]:
 78    """Generate unique Languages by randomly sampling subsets of Expressions, either in a uniform or stratified way.
 79    If there are fewer than `sample_size` possible Languages up to size `max_size`,
 80    this method will just return all languages up to that size (and so the sample may
 81    be smaller than `sample_size`).
 82
 83    Some use cases:
 84
 85    With `sample_size=None`, get all languages.
 86    >>> random_languages(expressions)
 87
 88    With `sample_size` and uniform sampling, get random languages:
 89    >>> random_languages(expressions, sample_size=1000)
 90
 91    Stratified sampling, with and without a `max_size`:
 92    >>> random_languages(expressions, sample_size=1000, sampling_strategy="stratified")
 93    >>> random_languages(expressions, sample_size=1000, sampling_strategy="stratified", max_size=10)
 94
 95    Args:
 96        expressions: all possible expressions
 97        sampling_strategy: how to sample subsets of expressions
 98            uniform: for every expression, choose whether or not to include it in a given language
 99            stratified: first sample a size for a Language, then choose that many random Expressions
100                (i) this has the effect of "upsampling" from smaller Language sizes
101                (ii) this can be used with `max_size` to only generate Languages up to a given number of expressions
102        sample_size: how many languages to return
103            if None, will return all languages up to `max_size`
104        language_class: type of Language
105        max_size: largest possible Language to generate
106            if None, will be the length of `expressions`
107            NB: this argument has no effect when `sampling_strategy` is "uniform"
108
109    Returns:
110        a list of randomly sampled Languages
111    """
112    # TODO: update docstring
113    if sampling_strategy not in ("uniform", "stratified"):
114        raise ValueError("Only 'uniform' and 'stratified' sampling are supported.")
115    expressions = list(expressions)
116    num_expr = len(expressions)
117    if max_size is None:
118        max_size = num_expr
119    num_subsets = upto_comb(num_expr, max_size)
120    if sample_size is None or num_subsets < sample_size:
121        print(f"Due to argument combination, returning all languages.")
122        return list(
123            all_languages(expressions, language_class=language_class, max_size=max_size)
124        )
125    languages = []
126    subsets = set()
127    while len(languages) < sample_size:
128        if sampling_strategy == "stratified":
129            lang_size = random.randint(1, max_size)
130            expr_indices = tuple(sorted(random.sample(range(num_expr), lang_size)))
131        elif sampling_strategy == "uniform":
132            expr_indices = tuple(
133                [idx for idx in range(num_expr) if random.choice((True, False))]
134            )
135        if expr_indices not in subsets:
136            subsets.add(expr_indices)
137            languages.append(language_class([expressions[idx] for idx in expr_indices]))
138    return languages

141def generate_languages(
142    language_class: Type[Language],
143    expressions: list[Expression],
144    lang_size: int,
145    sample_size: int,
146    criterion: Callable[[Expression], bool] = lambda *_: True,
147    fixed_wordcount=False,
148    dummy_name="sampled_lang_",
149    id_start: int = 0,
150    exact_sample=False,
151    verbose=False,
152) -> dict[str, Any]:
153    """Generate languages by randomly sampling vocabularies as bags of expressions.
154
155    A predicate (binary-valued property) of expressions may be supplied, which can be used to adjust the composition of vocabularies (e.g., by the percent of expressions satisfying the predicate).
156
157    If sample size <= nCr, then take a random sample_size set of combinations. Otherwise, to prevent repeat languages, treat nCr as the sample size.
158
159    Args:
160        expressions: a list of the possible expressions to sample from.
161
162        lang_size: the maximum (or exact) number of expressions in each language.
163
164        sample_size: the number of languages to generate.
165
166        criterion: the predicate, (e.g. semantic universal) by which to split the expressions into those satisfying and those not, and then sample languages with degrees of naturalness based on the percentage from those satisfying. Must apply at the expression level. By default is a trivial criterion, so that all expressions are 'quasi-natural'.
167
168        fixed_wordcount: whether to vary the language size from 1 to lang_size.
169
170        verbose: How detailed the progress of sampling should be, printed to stdout.
171
172        dummy_name: the default name to give to each sampled language, e.g. `sampled_lang_42`. These should not collide with any actual natural language names if the efficient communication experiment does use natural language data.
173
174        id_start: an integer representing the number of languages already generated in an experiment. Languages sampled will be named according to this number. For example, if id_start is 0, the first language sampled will be named `sampled_lang_0`. Note that the largest id does not necessarily track the actual size of languages saved for the experiment, but it does track how many languages have been generated.
175
176        exact_sample: a boolean representing whether to sample until the exact sample size is filled. If True, the resulting pool of languages may not be unique.
177
178        verbose: a boolean representing how verbose output should be during sampling.
179
180    Returns:
181        a dict representing the generated pool of languages and the updated id_start, of the form
182
183            {
184                "languages": (list of updated languages)
185                "id_start": (updated length of languages)
186            }
187
188    Examples:
189
190        >>> # Turn the knob on a universal property for modals
191        >>> expressions = load_expressions(expressions_file)
192        >>> universal_property = iff
193        >>> result = generate_languages(
194        ...    ModalLanguage,
195        ...    expressions,
196        ...    lang_size,
197        ...    sample_size,
198        ...    universal_property,
199        ...)
200        >>> languages = result["languages"]
201        >>> id_start = result["id_start"]
202
203    """
204    # split the expressions
205    natural_terms = []
206    unnatural_terms = []
207    for x in expressions:
208        natural_terms.append(x) if criterion(x) else unnatural_terms.append(x)
209
210    word_amounts = [lang_size] if fixed_wordcount else range(1, lang_size + 1)
211    total_word_amount = len(expressions)
212    word_amt_sample_size = int(sample_size / lang_size)
213
214    expressions_indices = list(range(total_word_amount))
215    languages = set()
216
217    # For each language size
218    for word_amount in word_amounts:
219        # If sample size > all possible languages (of any degree), just generate the latter.
220        if word_amt_sample_size > comb(total_word_amount, word_amount):
221            if verbose:
222                print(
223                    f"Enumerating {word_amt_sample_size} languages of size {word_amount}"
224                )
225            result = enumerate_all_languages(
226                language_class,
227                id_start,
228                expressions,
229                expressions_indices,
230                word_amount,
231                dummy_name=dummy_name,
232            )
233            enumerated_langs = result["languages"]
234            languages = languages.union(enumerated_langs)
235            id_start = result["id_start"]
236
237        # Otherwise, take random sample
238        else:
239            if verbose:
240                print(
241                    f"Generating {word_amt_sample_size} languages of size {word_amount}"
242                )
243            result = sample_quasi_natural(
244                language_class=language_class,
245                natural_terms=natural_terms,
246                unnatural_terms=unnatural_terms,
247                lang_size=word_amount,
248                sample_size=word_amt_sample_size,
249                id_start=id_start,
250                dummy_name=dummy_name,
251                verbose=verbose,
252            )
253
254            rlangs = result["languages"]
255            id_start = result["id_start"]
256            languages = languages.union(rlangs)
257
258    if exact_sample:
259        # Randomly choose a lang size and continue sampling until sample_size achieveds
260        additional_sample = sample_size - len(languages)
261        if verbose:
262            print(f"Sampled {len(languages)} out of {sample_size} languages.")
263        while additional_sample > 0:
264            word_amount = random.choice(word_amounts)
265            if verbose:
266                print(
267                    f"Filling remaining languages by sampling {additional_sample} languages of size {word_amount}"
268                )
269
270            rlangs = sample_quasi_natural(
271                language_class,
272                natural_terms,
273                unnatural_terms,
274                word_amount,
275                additional_sample,
276                id_start,
277                dummy_name=dummy_name,
278                verbose=verbose,
279            )
280            languages = languages.union(rlangs)
281            additional_sample = sample_size - len(languages)
282            print(additional_sample)
283
284    return {"languages": list(languages)[:sample_size], "id_start": id_start}

335def sample_quasi_natural(
336    language_class: Type[Language],
337    natural_terms: list[Expression],
338    unnatural_terms: list[Expression],
339    lang_size: int,
340    sample_size: int,
341    id_start: int,
342    dummy_name="sampled_lang_id",
343    verbose=False,
344) -> dict[str, Any]:
345    """Turn the knob on degree quasi-naturalness for a sample of languages, either by enumerating or randomly sampling unique subsets of all possible combinations.
346
347    Args:
348        natural_terms: expressions satisfying some criteria of quasi-naturalness, e.g, a semantic universal.
349
350        unnatural_terms: expressions not satisfying the criteria.
351
352        lang_size: the exact number of expressions a language must have.
353
354        sample_size: how many languages to sample.
355
356    Returns:
357        a dict containing the randomly sampled quasi-natural languages and the updated id_start, of the form
358
359            {
360                "languages": (list of updated languages)
361                "id_start": (updated length of languages)
362            }
363    """
364    languages = set()
365
366    natural_indices = list(range(len(natural_terms)))
367    unnatural_indices = list(range(len(unnatural_terms)))
368
369    # by default, expresions:= natural_terms, i.e. all degree-1.0
370    degrees = [lang_size]
371    if unnatural_terms:
372        degrees = list(range(lang_size + 1))
373    degree_sample_size = int(np.ceil(sample_size / len(degrees)))
374
375    # For each fraction of the lang size
376    for num_natural in tqdm(degrees):
377        num_unnatural = 0
378        if unnatural_terms:
379            num_unnatural = lang_size - num_natural
380
381        # If sample size > possible languages, just generate the latter.
382        possible_langs = comb(len(natural_terms), num_natural) * comb(
383            len(unnatural_terms), num_unnatural
384        )
385        if not possible_langs:
386            raise ValueError(
387                f"combinations is 0: check comb({len(natural_terms)}, {num_natural}) * comb({len(unnatural_terms)}, {num_unnatural})"
388            )
389
390        if degree_sample_size > possible_langs:
391            if verbose:
392                print(
393                    f"Enumerating {possible_langs} for degree {num_natural/lang_size}"
394                )
395            result = enumerate_all_languages(
396                language_class,
397                id_start,
398                natural_terms,
399                natural_indices,
400                num_natural,
401                unnatural_terms,
402                unnatural_indices,
403                num_unnatural,
404            )
405            enumerated_langs = result["languages"]
406            languages = languages.union(enumerated_langs)
407            id_start = result["id_start"]
408
409        # Otherwise, take a random sample
410        else:
411            if verbose:
412                print(
413                    f"Sampling {degree_sample_size} languages of size {lang_size} with degree {num_natural/lang_size}"
414                )
415
416            # Sample unique languages
417            seen = set()
418            for _ in range(degree_sample_size):
419                vocabulary = random_combination_vocabulary(
420                    seen, num_natural, natural_terms, num_unnatural, unnatural_terms
421                )
422                id_start += 1
423                language = language_class(
424                    vocabulary, name=rename_id(dummy_name, id_start)
425                )
426                languages.add(language)
427
428    assert len(languages) == len(set(languages))
429    return {"languages": languages, "id_start": id_start}

442def enumerate_all_languages(
443    language_class: Type[Language],
444    id_start: int,
445    natural_terms: list[Expression],
446    natural_indices: list[int],
447    num_natural: int = 0,
448    unnatural_terms: list[Expression] = [],
449    unnatural_indices: list[int] = [],
450    num_unnatural: int = 0,
451    dummy_name="sampled_lang_id",
452    verbose=False,
453) -> dict[str, Any]:
454    """When the sample size requested is greater than the size of all possible languages, just enumerate all the possible languages.
455
456    Args:
457        language_class: the kind of Language to construct
458
459        id_start: a number to start counting from for assigning names with numerical ids to languages.
460
461        natural_indices: the indices of quasi-natural languages already seen
462
463        num_natural: the number of quasi-natural languages to sample
464
465        natural_terms: the list of quasi-natural terms to sample from
466
467        unnatural_indices: the indices of non-quasi-natural languages already seen
468
469        num_unnatural: the number of non-quasi-natural languages to sample; 0 by default
470
471        unnatural_terms: the list of non-quasi-natural terms to sample from; empty by default.
472
473        dummy_name: the format of the string to name each language constructed.
474
475    Returns:
476        a dict containing a set of languages and the updated id_start,  of the form
477
478            {
479                "languages": (list of updated languages)
480                "id_start": (updated length of languages)
481            }
482    """
483    # combinations is invariant to order
484    natural_subsets = list(combinations(natural_indices, num_natural))
485    unnatural_subsets = list(combinations(unnatural_indices, num_unnatural))
486
487    languages = set()
488    # Construct the languages
489    for natural_subset in natural_subsets:
490        for unnatural_subset in unnatural_subsets:
491            vocabulary = [natural_terms[idx] for idx in natural_subset] + [
492                unnatural_terms[idx] for idx in unnatural_subset
493            ]
494            id_start += 1
495            language = language_class(vocabulary, name=rename_id(dummy_name, id_start))
496            languages.add(language)
497    return {"languages": languages, "id_start": id_start}

500def random_combination_vocabulary(
501    seen: set,
502    num_natural: int,
503    natural_terms: list[Expression],
504    num_unnatural: int = 0,
505    unnatural_terms: list[Expression] = [],
506) -> list[Language]:
507    """Get a single vocabulary for a specific language size by choosing a random combination of natural and unnatural terms.
508
509    Args:
510        seen: the list of language indices already seen
511
512        num_natural: int
513
514        natural_terms: list[Expression]
515
516        num_unnatural: int=0
517
518        unnatural_terms: list[Expression]=[]
519
520    Returns:
521        languages: the extended list of input languages.
522    """
523    while True:
524        nat_sample_indices = tuple(
525            sorted(random.sample(range(len(natural_terms)), num_natural))
526        )
527        unnat_sample_indices = ()
528        if unnatural_terms:
529            unnat_sample_indices = tuple(
530                sorted(random.sample(range(len(unnatural_terms)), num_unnatural))
531            )
532        sample_indices = (nat_sample_indices, unnat_sample_indices)
533        if sample_indices not in seen:
534            # keep track of languages chosen
535            seen.add(sample_indices)
536
537        # Add language
538        vocabulary = [natural_terms[idx] for idx in nat_sample_indices] + [
539            unnatural_terms[idx] for idx in unnat_sample_indices
540        ]
541        break
542    return vocabulary
